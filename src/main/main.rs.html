36pub struct MyScript {
37    is_downed: bool,
38}
39
40impl Script for MyScript {
41    fn new(is_downed: bool) -> MyScript {
42        MyScript { is_downed }
43    }
44
45    fn action(&mut self, game_object: &mut GameObject) {
46        if !self.is_downed {
47            game_object.position = Position {
48                x: game_object.position.x,
49                y: game_object.position.y - 1,
50                z: game_object.position.z,
51                is_relative: game_object.position.is_relative,
52            };
53            self.is_downed = true;
54        } else {
55            game_object.position = Position {
56                x: game_object.position.x,
57                y: game_object.position.y + 1,
58                z: game_object.position.z,
59                is_relative: game_object.position.is_relative,
60            };
61            self.is_downed = false;
62        }
63    }
64}
