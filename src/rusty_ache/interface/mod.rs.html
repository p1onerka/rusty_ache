1//! Helper utilities for creating game objects, initializing scenes, and setting up the game engine.
2//!
3//! This module provides convenience functions to build vectors of game objects from simplified
4//! image-based object definitions (`ObjectWithImage`), initialize scenes with those objects,
5//! and create configured instances of the game engine.
6//!
7//! These functions support workflow from asset loading to scene setup to engine initialization.
8

23/// Represents a simplified interface to a game object with associated image and placement.
24///
25/// Encapsulates the image filepath and positional attributes along with shadow flag
26/// used for sprite rendering.
27#[derive(Clone)]
28pub struct ObjectWithImage<'a> {
29    /// The filepath to the sprite's image.
30    image_path: &'a str,
31    /// The x-coordinate position in the game world.
32    x: i32,
33    /// The y-coordinate position in the game world.
34    y: i32,
35    /// Whether the sprite should cast a shadow.
36    has_shadow: bool,
37}
38
39/// Converts a slice of `ObjectWithImage` entries into a vector of fully constructed `GameObject`s.
40///
41/// Assigns z-coordinates incrementally to arrange objects from farthest (z=1) to closest.
42/// Loads sprite images from their file paths and creates corresponding sprite components.
43///
44/// # Parameters
45/// - `objs`: Slice of simplified object image descriptions.
46///
47/// # Returns
48/// Vector of fully constructed game objects ready for scene insertion.
49pub fn create_gameobj_vec(objs: &[ObjectWithImage]) -> Vec<GameObject> {
50    let mut res = Vec::new();
51    let mut z_coord = 1;
52    for obj in objs {
53        res.push(GameObject::new(
54            vec![Box::new(Sprite::new(
55                Some(ImageReader::open(obj.image_path).unwrap().decode().unwrap()),
56                obj.has_shadow,
57                (0, 0),
58            ))],
59            None,
60            Position {
61                x: obj.x,
62                y: obj.y,
63                z: z_coord,
64                is_relative: false,
65            },
66        ));
67        z_coord += 1;
68    }
69    res
70}

72/// Creates an `ObjectWithImage` instance from image path and position data.
73///
74/// Encapsulates the essential information needed to create a game object sprite.
75///
76/// # Parameters
77/// - `image_path`: File path to the sprite image.
78/// - `x`, `y`: Position coordinates.
79/// - `has_shadow`: Whether the sprite casts shadow.
80///
81/// # Returns
82/// A new `ObjectWithImage` instance.
83pub fn create_obj_with_img(image_path: &str, x: i32, y: i32, has_shadow: bool) -> ObjectWithImage {
84    ObjectWithImage {
85        image_path,
86        x,
87        y,
88        has_shadow,
89    }
90}

92/// Initializes a `Scene` from a slice of background objects and a single main object.
93///
94/// Converts all background objects into game objects with sprites and setups the main object
95/// with its sprite and offset.
96///
97/// # Parameters
98/// - `objs`: Slice of background `ObjectWithImage`.
99/// - `main_obj`: The main object displayed in the scene.
100///
101/// # Returns
102/// A full `Scene` instance initialized and ready for rendering.
103pub fn init_scene(objs: &[ObjectWithImage], main_obj: ObjectWithImage) -> Scene {
104    let game_objs = create_gameobj_vec(objs);
105    Scene::new(
106        game_objs,
107        vec![Box::new(Sprite::new(
108            Some(
109                ImageReader::open(main_obj.image_path)
110                    .unwrap()
111                    .decode()
112                    .unwrap(),
113            ),
114            true,
115            (60, -60),
116        ))],
117        Position {
118            x: main_obj.x,
119            y: main_obj.y,
120            z: 0,
121            is_relative: false,
122        },
123    )
124}
125
126/// Creates and initializes a `GameEngine` instance using the given scene and resolution.
127///
128/// Wraps configuration creation and engine construction in one step.
129///
130/// # Parameters
131/// - `scene`: The initial scene the engine will manage.
132/// - `width`, `height`: Screen resolution dimensions.
133///
134/// # Returns
135/// A fully initialized `GameEngine` ready to run.
136pub fn init_engine(scene: Scene, width: u32, height: u32) -> GameEngine {
137    GameEngine::new(
138        Box::new(EngineConfig::new(Resolution::new(width, height))),
139        scene,
140    )
141}

144mod tests {

147    use super::*;

149    #[test]
150    fn test_create_obj_with_img() {
151        let obj = create_obj_with_img("./image", 100, 100, true);
152        assert_eq!(obj.image_path, "./image");
153        assert_eq!(obj.x, 100);
154        assert_eq!(obj.y, 100);
155        assert!(obj.has_shadow);
156    }
157
158    #[test]
159    fn test_create_gameobj_vec() {
160        let objs = [create_obj_with_img(
161            "./resources/perf_diag.png",
162            200,
163            200,
164            false,
165        )];
166        let owi = create_gameobj_vec(&objs);
167        assert_eq!(owi.len(), objs.len());
168        assert_eq!(objs[0].x, owi[0].position.x);
169        assert_eq!(objs[0].y, owi[0].position.y);
170    }
171
172    #[test]
173    fn test_init_scene() {
174        let objs = [create_obj_with_img(
175            "./resources/perf_diag.png",
176            200,
177            200,
178            false,
179        )];
180        let main_obj = create_obj_with_img("./resources/perf_diag.png", 300, 300, true);
181        let main_obj_x = main_obj.x;
182        let main_obj_y = main_obj.y;
183        let scene = init_scene(&objs, main_obj);
184        assert_eq!(scene.main_object.position.x, main_obj_x);
185        assert_eq!(scene.main_object.position.y, main_obj_y);
186    }
