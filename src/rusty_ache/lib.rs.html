10/// Represents screen resolution.
11///
12/// Holds the width and height in pixels, encapsulating display dimensions.
13#[derive(Copy, Clone)]
14pub struct Resolution {
15    /// Width of the screen or rendering area in pixels.
16    width: u32,
17    /// Height of the screen or rendering area in pixels.
18    height: u32,
19}
20
21impl Resolution {
22    /// Creates a new `Resolution` instance with specified width and height.
23    ///
24    /// # Parameters
25    /// - `width`: Width in pixels.
26    /// - `height`: Height in pixels.
27    ///
28    /// # Returns
29    /// A new `Resolution` struct with given dimensions.
30    pub fn new(width: u32, height: u32) -> Self {
31        Self { width, height }
32    }
33}
