1//! A trait describing a game object component entity, such as Sprite, Camera, etc.
2//!
3//! This module defines the core Component trait representing various components
4//! that can be attached to game entities. It also defines the component types
5//! and error types related to component handling.
6
7use image::DynamicImage;
8use std::any::Any;
9use std::fmt::Debug;
10
11pub mod script;
12pub mod sprite;
13mod velocity;
14
15/// Errors that can occur when handling components.
16pub enum ComponentError {
17    /// Component already exists.
18    Exist(Box<dyn Component>),
19    /// Operation could not be applied on component.
20    CannotApply(String),
21    /// An unknown error occurred.
22    UnknownError(String),
23    /// Invalid index was provided.
24    InvalidIndex(String),
25}
26
27/// Enum identifying types of components supported.
28#[derive(Debug, Eq, PartialEq, Clone)]
29pub enum ComponentType {
30    Sprite,
31    Velocity,
32    Action,
33}
34
35/// Trait that defines behavior of any game component.
36///
37/// Components represent modular features or data attached to an entity,
38/// such as a visual sprite, motion velocity, or behavior scripts.
39///
40/// Implementors must specify their component type and can optionally
41/// override accessors to retrieve related data.
42pub trait Component: Any {
43    /// Returns the component as a dynamic Any reference, allowing downcasting.
44    fn as_any(&self) -> &dyn Any;
45
46    /// Returns the component's type identifier.
47    fn get_component_type(&self) -> ComponentType;
48
49    /// Returns the sprite image if available.
50    ///
51    /// Default returns None; override in Sprite component implementations.
52    fn get_sprite_unchecked(&self) -> &Option<DynamicImage> {
53        &None
54    }
55
56    /// Indicates whether the component casts shadows.
57    ///
58    /// Default returns true; override if shadow behavior differs.
59    fn get_shadow_unchecked(&self) -> bool {
60        true
61    }
62
63    /// Returns the sprite offset if applicable.
64    ///
65    /// Default returns None; override if component supports sprite offsetting.
66    fn get_sprite_offset_unchecked(&self) -> Option<(i32, i32)> {
67        None
68    }
69}
