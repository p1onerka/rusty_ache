1//! A trait describing behavior of a script component in the game engine.
2//!
3//! Defines contract for game logic scripts that operate on game objects.
4//! Each script can apply an action to modify a given `GameObject`.
5//! Also requires an associated constructor method with initialization state.
6
7use crate::engine::scene::game_object::GameObject;
8
9/// Trait representing a script that can be attached to a game object.
10///
11/// Scripts encapsulate game logic that modifies the game object's state.
12/// This trait requires the implementation of:
13/// - an `action` method performing operations on a mutable reference to `GameObject`
14/// - a `new` constructor with an initialization parameter `is_downed`
15///
16/// Implementors can define custom behavior for initialization and per-frame updates.
17pub trait Script {
18    /// Apply the script's action on the given mutable game object reference.
19    ///
20    /// Allows modifying object state, trigger events, or update components.
21    fn action(&mut self, game_object: &mut GameObject);
22
23    /// Construct a new instance of the script.
24    ///
25    /// The `is_downed` parameter allows differentiating initial states or modes.
26    /// Implementations can use this to customize setup logic.
27    fn new(is_downed: bool) -> Self
28    where
29        Self: Sized;
30}
